function ValidateEmail(mail) {
                if (/^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/.test(mail)) {
                    return (true)
                }
                return (false)
            }

            $('#callback1 .feedback-submit-button').click(function(e) {
                let mail = $('#feedback_models_FeedbackModel_mail').val();
                if (!ValidateEmail(mail)) {
                    $('#feedback_models_FeedbackModel_mail').addClass('reject');
                    return false;
                } else {
                    $('#feedback_models_FeedbackModel_mail').removeClass('reject');
                }
            });
