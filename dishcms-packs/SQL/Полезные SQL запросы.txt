Получение размера таблиц в базе данных (https://stackoverflow.com/a/9620273)
SELECT table_name AS `Table`, round(((data_length + index_length) / 1024 / 1024), 2) `Size in MB` FROM information_schema.TABLES WHERE table_schema = "имябазыданных";

Получение уникального списка ЧПУ с инкрементом
SET @i:=0; SET @a:=''; SELECT IF(@a=`alias`,CONCAT(`alias`,'-',@i:=@i+1),IF(@i:=0,0,@a:=alias)) AS a from mytable

SET @i:=0; SET @a:=''; UPDATE `product` SET `alias`=IF(@a=`alias`,CONCAT(`alias`,'-',@i:=@i+1),IF(@i:=0,0,@a:=`alias`)) ORDER BY `alias`;
