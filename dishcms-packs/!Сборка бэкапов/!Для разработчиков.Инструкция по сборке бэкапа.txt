$installDataBaseFile = dirname(__FILE__) . '/database.sql';
if (is_file($installDataBaseFile)) {
    error_reporting(E_NONE);
    ini_set('display_errors', 0);
    set_time_limit(0);
    $dbconfig = dirname(__FILE__) . '/protected/config/db.php';
	echo 'Не обновляйте страницу, идет импорт данных в базу данных<br/>';
    if ( is_file($dbconfig) ) {
        $db = include($dbconfig);
        if( isset($db['connectionString']) && isset($db['username']) && isset($db['password']) ) {
            $dbhost = preg_replace('/^.*host=([^;]+);.*$/', '$1', $db['connectionString']);
            $dbname = preg_replace('/^.*dbname=(.*)$/', '$1', $db['connectionString']);
            $mysqli = new mysqli($dbhost, $db['username'], $db['password'], $dbname);
            if (!$mysqli->connect_errno) {
                $result = $mysqli->query('SHOW TABLES');
                if($result->num_rows > 0) {
                    echo '<b>ОШИБКА!</b> База данных содержит таблицы. Очистите базу данных перед установкой!';
                }
                else {
                    exec("mysql -h {$dbhost} -u {$db['username']} " . ($db['password'] ? " -p {$db['password']}" : '') . " {$dbname} < {$installDataBaseFile}");

                    if(!@unlink($installDataBaseFile)) {
                        echo 'База данных успешно импортирована.<br/><b>ОШИБКА!</b> Файл /database.sql не был удален. Удалите файл вручную!';
                    }
                    else {
                        header("HTTP/1.1 301 Moved Permanently");
                        header('Location: /');
                    }
                }
                exit;
            }
        }
    }
    echo 'Не удается подключиться к базе данных';
    exit;
}
